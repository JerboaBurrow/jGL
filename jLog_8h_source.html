50 inline std::string get_time()
51 {
52 time_t time = std::time(nullptr);
53 char time_buf[80];
54 struct tm ts;
55 ts = *localtime(&time);
56 strftime(time_buf, sizeof(time_buf), "%a %Y-%m-%d %H:%M:%S %Z", &ts);
57
58 return std::string(time_buf);
